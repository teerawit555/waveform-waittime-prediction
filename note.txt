Train

python scripts/extract_features.py --mode train --in data/data_for_train_m_n.csv --out data/processed/train_features.csv
python scripts/make_wave_tensor.py --in data/data_for_train_m_n.csv --out data/processed/train_wave_tensor.npz --target-len 1000 --label-col wait_time_ms
python scripts/train_tcn_encoder.py --waves data/processed/train_wave_tensor.npz --out models/tcn_v1 --epochs 30 --batch-size 64 --lr 0.001 --embedding-dim 64 --log-target
python scripts/export_tcn_encoder.py --model models/tcn_v1 --waves data/processed/train_wave_tensor.npz --out data/processed/train_tcn_embed.csv
python scripts/merge_features_and_embeddings.py --features data/processed/train_features.csv --embeddings data/processed/train_tcn_embed.csv --out data/processed/train_hybrid.csv
python scripts/train_ag_1stage.py --data data/processed/train_hybrid.csv --label wait_time_ms --model-dir AutogluonModels/ag_1stage_hybrid_v1 --time-limit 300 --log-target


Predict

python scripts/extract_features.py --mode pred --in data/data_for_pred_m_n.csv --out data/processed/infer_features.csv
python scripts/make_wave_tensor.py --in data/data_for_pred_m_n.csv --out data/processed/infer_wave_tensor.npz --target-len 1000
python scripts/export_tcn_encoder.py --model models/tcn_v1 --waves data/processed/infer_wave_tensor.npz --out data/processed/infer_tcn_embed.csv
python scripts/merge_features_and_embeddings.py --features data/processed/infer_features.csv --embeddings data/processed/infer_tcn_embed.csv --out data/processed/infer_hybrid.csv
python scripts/predict_ag_1stage.py --model-path AutogluonModels/ag_1stage_hybrid_v1 --in data/processed/infer_hybrid.csv --out data/processed/prediction/pred_1stage_hybrid.csv
